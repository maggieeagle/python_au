56. Merge Intervals
https://leetcode.com/problems/merge-intervals/
class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort()

        if (intervals == []):
            return intervals

        begin_prev = intervals[0][0]
        end_prev = intervals[0][1]

        ans: List[List[int]] = []
        for i in intervals:
            if i[0] > end_prev:
                ans.append([begin_prev, end_prev])
                begin_prev = i[0]
                end_prev = i[1]
            else:
                end_prev = max(end_prev, i[1])
        ans.append([begin_prev, end_prev])
        return ans
