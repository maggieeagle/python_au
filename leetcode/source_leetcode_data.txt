509. Fibonacci Number
https://leetcode.com/problems/fibonacci-number/
class Solution:
    def fib(self, N: int) -> int:
        prevprev = 0
        prev = 1
        fi = 0
        if N == 0:
            return 0
        if N == 1:
            return 1
        for i in range(N - 1):
            fi = prevprev + prev
            prevprev = prev
            prev = fi
        return fi
