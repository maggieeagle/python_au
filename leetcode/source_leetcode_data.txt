9. Palindrome Number
https://leetcode.com/problems/palindrome-number/
class Solution:
    def isPalindrome(self, x: int) -> bool:
        symbols = list(str(x))
        length = len(str(x))
        while length > 0:
            if symbols[0] != symbols[length - 1]:
                return False
            if length <= 1:
                return True
            del symbols[0]
            del symbols[length - 2]
            length -= 2
        return True
