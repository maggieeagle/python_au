69. Sqrt(x)
https://leetcode.com/problems/sqrtx/
class Solution:
    def mySqrt(self, x: int) -> int:
        if x == 1:
            return 1
        first = 0
        last = x
        while True:
            middle = (first + last)/2
            if x > middle**2:
                first = middle
            if x < middle**2:
                last = middle
            if x == middle**2:
                return int(middle)
            if abs(middle**2 - x) < 0.1:
                print(middle)
                if round(middle)**2 > x:
                    return int(middle)
                else:
                    return int(round(middle))
